    <script>
        // --- PRO SENSORY ENGINE: MECHANICAL "THOCK" ---
        const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
        let ambientSource = null;
        let ambientGain = audioCtx.createGain();
        ambientGain.connect(audioCtx.destination);
        let isMuted = false;
        let wakeLock = null;

        const AMBIENT_VOL = 0.15; 
        const UI_THOCK_VOL = 0.4; // Decreased but still thocky

        function playThock() {
            if (audioCtx.state === 'suspended') audioCtx.resume();
            
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            
            osc.type = 'triangle';
            osc.frequency.setValueAtTime(180, audioCtx.currentTime);
            osc.frequency.exponentialRampToValueAtTime(40, audioCtx.currentTime + 0.1);
            
            gain.gain.setValueAtTime(UI_THOCK_VOL, audioCtx.currentTime);
            gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.1);
            
            osc.connect(gain);
            gain.connect(audioCtx.destination);
            
            osc.start();
            osc.stop(audioCtx.currentTime + 0.1);
        }

        function playZenBell() {
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.type = 'sine';
            osc.frequency.setValueAtTime(880, audioCtx.currentTime); 
            osc.frequency.exponentialRampToValueAtTime(440, audioCtx.currentTime + 2);
            gain.gain.setValueAtTime(0.5, audioCtx.currentTime);
            gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 3);
            osc.connect(gain);
            gain.connect(audioCtx.destination);
            osc.start();
            osc.stop(audioCtx.currentTime + 3);
        }

        function startAmbient() {
            if (ambientSource) return;
            const bufferSize = 4096;
            ambientSource = audioCtx.createScriptProcessor(bufferSize, 1, 1);
            ambientSource.onaudioprocess = function(e) {
                const output = e.outputBuffer.getChannelData(0);
                for (let i = 0; i < bufferSize; i++) { output[i] = Math.random() * 2 - 1; }
            };
            const filter = audioCtx.createBiquadFilter();
            filter.type = 'lowpass';
            filter.frequency.value = 350; 
            ambientSource.connect(filter);
            filter.connect(ambientGain);
            ambientGain.gain.setValueAtTime(0, audioCtx.currentTime);
        }

        async function requestWakeLock() {
            try {
                if ('wakeLock' in navigator) {
                    wakeLock = await navigator.wakeLock.request('screen');
                }
            } catch (err) { console.log("WakeLock failed."); }
        }

        const ring = document.getElementById('ring');
        const timeText = document.getElementById('time-text');
        const actionBtn = document.getElementById('action-btn');
        const exitBtn = document.getElementById('exit-btn');
        const strictToggle = document.getElementById('strict-toggle');
        const strictLabel = document.getElementById('strict-label');
        const appFrame = document.getElementById('app-frame');
        const container = document.getElementById('interaction-area');
        const modeSelect = document.getElementById('mode-select');
        const muteToggle = document.getElementById('mute-toggle');

        const radius = 135;
        const centerX = 155;
        const centerY = 155;
        const circumference = 2 * Math.PI * radius;

        let remainingSeconds = 1500;
        let isRunning = false;
        let isDragging = false;
        let countdown;

        muteToggle.addEventListener('click', () => {
            isMuted = !isMuted;
            muteToggle.innerText = isMuted ? "🔇" : "🔊";
            if (isRunning) {
                ambientGain.gain.setTargetAtTime(isMuted ? 0 : AMBIENT_VOL, audioCtx.currentTime, 0.5);
            }
            playThock();
        });

        // --- STRICT MODE RED ALERT LOGIC ---
        strictToggle.addEventListener('change', () => {
            playThock();
            if(strictToggle.checked) {
                strictLabel.classList.add('strict-active');
            } else {
                strictLabel.classList.remove('strict-active');
            }
        });

        function updateUI(seconds) {
            const progressRatio = seconds / 3600; 
            const offset = circumference - (progressRatio * circumference);
            ring.style.strokeDashoffset = offset;
            const m = Math.floor(seconds / 60);
            const s = seconds % 60;
            timeText.innerText = `${m.toString().padStart(2, '0')}:${s.toString().padStart(2, '0')}`;
        }

        function handleInteraction(e) {
            if (isRunning) return;
            const rect = container.getBoundingClientRect();
            const clientX = e.clientX || (e.touches ? e.touches[0].clientX : 0);
            const clientY = e.clientY || (e.touches ? e.touches[0].clientY : 0);
            const x = clientX - rect.left - centerX;
            const y = clientY - rect.top - centerY;
            let angle = Math.atan2(y, x) + Math.PI / 2;
            if (angle < 0) angle += 2 * Math.PI;
            const minutes = Math.round((angle / (2 * Math.PI)) * 60);
            
            if (minutes * 60 !== remainingSeconds) playThock(); // Mechanical thock on interaction
            
            remainingSeconds = minutes * 60;
            ring.style.transition = 'none';
            updateUI(remainingSeconds);
        }

        container.addEventListener('mousedown', (e) => { isDragging = true; handleInteraction(e); });
        window.addEventListener('mouseup', () => { isDragging = false; });
        window.addEventListener('mousemove', (e) => { if (isDragging) handleInteraction(e); });
        container.addEventListener('touchstart', (e) => { isDragging = true; handleInteraction(e); });
        window.addEventListener('touchend', () => { isDragging = false; });
        window.addEventListener('touchmove', (e) => { if (isDragging) handleInteraction(e); });

        actionBtn.addEventListener('click', () => {
            playThock();
            if (!isRunning) {
                if (remainingSeconds <= 0) return;
                isRunning = true;
                if (audioCtx.state === 'suspended') audioCtx.resume();
                
                startAmbient();
                requestWakeLock(); // Prevent sleep
                
                // Fade in Ambient Hum
                ambientGain.gain.setTargetAtTime(isMuted ? 0 : AMBIENT_VOL, audioCtx.currentTime, 1.5);
                
                ring.style.transition = 'stroke-dashoffset 1s linear';
                modeSelect.disabled = true;

                if (strictToggle.checked) {
                    actionBtn.innerText = "FOCUSING...";
                    actionBtn.disabled = true;
                    exitBtn.disabled = true;
                } else {
                    actionBtn.innerText = "PAUSE";
                }
                countdown = setInterval(() => {
                    remainingSeconds--;
                    updateUI(remainingSeconds);
                    if (remainingSeconds <= 0) {
                        clearInterval(countdown);
                        playZenBell();
                        if (wakeLock) wakeLock.release();
                        setTimeout(() => {
                            alert("Hurray! Session Completed 🎉");
                            location.reload();
                        }, 500);
                    }
                }, 1000);
            } else {
                // Pause Logic: Fade out hum and release wake lock
                clearInterval(countdown);
                isRunning = false;
                ambientGain.gain.setTargetAtTime(0, audioCtx.currentTime, 0.8);
                if (wakeLock) { wakeLock.release(); wakeLock = null; }
                actionBtn.innerText = "RESUME";
            }
        });

        exitBtn.addEventListener('click', () => {
            playThock();
            if (confirm("THINK TWICE ⚠️")) {
                if (wakeLock) wakeLock.release();
                location.reload();
            }
        });

        updateUI(1500);
    </script>
